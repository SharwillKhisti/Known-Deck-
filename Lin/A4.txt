## Q11 Arithmetic operations on integers


echo "Arithmetic Operations"
echo "Enter num1: "
read num1
echo "Enter num2: "
read num2
add=$((num1+num2))
sub=$((num1-num2))
mul=$((num1*num2))
div=$(expr $num1 / $num2)
expo=$((num1**num2))
echo "Addition is $add"
echo "Subtraction is $sub"
echo "Multiplication is $mul"
echo "Division is $div"
echo "Exponentiation is $expo"


## Q12 Arithmetic operations on float data


echo "Arithmetic Operations"
echo "Enter num1: "
read num1
echo "Enter num2: "
read num2
add=`echo $num1 + $num2 | bc`
sub=`echo $num1 - $num2 | bc`
mul=`echo "$num1 * $num2" | bc`
div=`echo $num1 / $num2 | bc`
expo=`echo $num1 ^ $num2 | bc`
echo "Addition is $add"
echo "Subtraction is $sub"
echo "Multiplication is $mul"
echo "Division is $div"
echo "Exponentiation is $expo"
